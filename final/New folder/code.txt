import networkx as nx
import pandas as pd
import random
import matplotlib.pyplot as plt
from littleballoffur import HybridNodeEdgeSampler

def load_edge_list(file_path):
    data_load=pd.read_csv(file_path,sep=',',names=['n1','n2'])
    return data_load

def dataframe_to_graph(df):
  g=nx.from_pandas_edgelist(df,'n1','n2',create_using=nx.Graph())
  return g

def rand(graph):
  x=random.choice(list(graph.nodes()))
  return x

def shortest_path(original_graph):
    sample_graph = nx.Graph()
    number_of_edges = int(0.2*original_graph.number_of_edges())
    while sample_graph.number_of_edges() < number_of_edges:
      source_node = rand(original_graph)
      target_node = rand(original_graph)
      while source_node == target_node:
         source_node = rand(original_graph)
         target_node = rand(original_graph)
      if nx.has_path(original_graph, source_node, target_node):
        all_shortest_paths = list(nx.all_shortest_paths(original_graph, source=source_node, target=target_node))
        for shortest_path_nodes in all_shortest_paths:
                sample_graph.add_nodes_from(shortest_path_nodes)
                for i in range(len(shortest_path_nodes)-1):
                    sample_graph.add_edge(shortest_path_nodes[i], shortest_path_nodes[i + 1])

      else:
          print(f"There is no path between {source_node} and {target_node} in the original graph.")

    return sample_graph


def sample_graph(org_graph):
    number_of_edges = int(0.2*org_graph.number_of_edges())
    sampler = HybridNodeEdgeSampler(number_of_edges = number_of_edges)
    new_graph = sampler.sample(org_graph)
    return new_graph

def largest_degree_nodes(graph, n):
    degrees = dict(graph.degree())
    sorted_degrees = sorted(degrees.items(), key=lambda x: x[1], reverse=True)
    array=sorted_degrees[:n]
    result_array = [tpl[0] for tpl in array]
    return result_array
